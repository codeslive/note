### 一、冒泡排序

```c
#include <stdio.h>
int main() {
	//冒泡排序 
	int num[5] = {1,5,3,9,7}; 
	int i,j;
	int temp;
	for(i=0;i<5-1;i++){
		for(j=0;j<5-1-i;j++){
			if(num[j]>num[j+1]){
				temp=num[j];
				num[j]=num[j+1];
				num[j+1]=temp;
			}
		}
	}
  
  	for(i=0;i<5;i++){
  		printf("%d ",num[i]);
	}
  
   return 0;

}


```

                break;

            if(i>=num){
                t=num;
                t_num=0;

//不允允许使用stract函数时
#include <stdio.h>

void strcat1(char *s1, char *s2){
    while(*s1!='\0')
        *s1++;
    while(*s2!='\0')
        *s1++ = *s2++;
    *s1='\0';
  
}

int main(){
  
    //定义两个字符数组
    char s1[20]="abcdefg";
    char s2[20]="hijklmn";
    //传参
    strcat1(s1,s2);
    puts(s1);
    return 0;
}
